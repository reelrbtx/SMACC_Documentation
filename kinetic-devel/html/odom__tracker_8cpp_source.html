151 bool OdomTracker::updateBackward(const nav_msgs::Odometry &odom)
152 {
153  // we initially accept any message if the queue is empty
155  geometry_msgs::PoseStamped base_pose;
156 
157  base_pose.pose = odom.pose.pose;
158  base_pose.header = odom.header;
159  baseTrajectory_.header = odom.header;
160 
161  bool acceptBackward = false;
162  bool pullingerror = false;
163  if (baseTrajectory_.poses.empty())
164  {
165  acceptBackward = false;
166  }
167  else
168  {
169  const geometry_msgs::Point &prevPoint = baseTrajectory_.poses.back().pose.position;
170  const geometry_msgs::Point &currePoint = base_pose.pose.position;
171  double lastpointdist = p2pDistance(prevPoint, currePoint);
172 
173  acceptBackward = !baseTrajectory_.poses.empty() && lastpointdist < minPointDistanceBackwardThresh_;
174 
175  pullingerror = lastpointdist > 2 * minPointDistanceBackwardThresh_;
176  }
177 
178  //ROS_INFO("Backwards, last distance: %lf < %lf accept: %d", dist, minPointDistanceBackwardThresh_, acceptBackward);
179  if (acceptBackward)
180  {
181  baseTrajectory_.poses.pop_back();
182  }
183  else if (pullingerror)
184  {
185  ROS_INFO_THROTTLE(2, "Incorrect backwards motion.");
186  }
187  else
188  {
190  }
191 
192  return acceptBackward;
193 }
199 bool OdomTracker::updateForward(const nav_msgs::Odometry &odom)
200 {
202  geometry_msgs::PoseStamped base_pose;
203 
204  base_pose.pose = odom.pose.pose;
205  base_pose.header = odom.header;
206  baseTrajectory_.header = odom.header;
207 
208  bool enqueueOdomMessage = false;

210  double dist = -1;

218  const geometry_msgs::Point &currePoint = base_pose.pose.position;
219  dist = p2pDistance(prevPoint, currePoint);
220  //ROS_WARN("dist %lf vs min %lf", dist, minPointDistanceForwardThresh_);
221 
222  if (dist > minPointDistanceForwardThresh_)
223  {
224  enqueueOdomMessage = true;

228  //ROS_WARN("skip odom, dist: %lf", dist);
229  enqueueOdomMessage = false;
230  }
231  }
232

235  baseTrajectory_.poses.push_back(base_pose);
236  }
237
