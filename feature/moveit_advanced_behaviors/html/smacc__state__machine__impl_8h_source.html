136  template <typename EventType>
137  void ISmaccStateMachine::postEvent(EventType *ev, EventLifeTime evlifetime)
138  {
139  if (evlifetime == EventLifeTime::CURRENT_STATE && (stateMachineCurrentAction == StateMachineInternalAction::STATE_EXITING ||
140  stateMachineCurrentAction == StateMachineInternalAction::TRANSITIONING))
141  {
142  ROS_WARN_STREAM("CURRENT STATE SCOPED EVENT SKIPPED, state is exiting/transitioning " << demangleSymbol<EventType>());
143  return;
144  // in this case we may lose/skip events, if this is not right for some cases we should create a queue
145  // to lock the events during the transitions. This issues appeared when a client asyncbehavior was posting an event
146  // meanwhile we were doing the transition, but the main thread was waiting for its correct finalization (with thread.join)
147  }
148 
149  std::unique_lock<std::recursive_mutex> lock(m_mutex_);
150 
151  if (lock.owns_lock())
152  {
153  // when a postting event is requested by any component, client, or client behavior
154  // we reach this place. Now, we propagate the events to all the state state reactors to generate
155  // some more events
156 
157  ROS_DEBUG_STREAM("[PostEvent entry point] " << demangleSymbol<EventType>());
158  auto currentstate = currentState_;
159  if (currentstate != nullptr)
160  {
161  propagateEventToStateReactors(currentstate, ev);
162  }
163 
164  this->signalDetector_->postEvent(ev);
165  }
166  else
167  {
168  ROS_WARN_STREAM("Event with lifetime (Current state) was discarded." << demangleSymbol<EventType>());
169  }
170  }

304  template <typename TSmaccSignal, typename TMemberFunctionPrototype, typename TSmaccObjectType>
305  boost::signals2::connection ISmaccStateMachine::createSignalConnection(TSmaccSignal &signal,
306  TMemberFunctionPrototype callback,
307  TSmaccObjectType *object)
308  {
309  static_assert(std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
310  std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
311  std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value ||
312  std::is_base_of<StateReactor, TSmaccObjectType>::value ||
313  std::is_base_of<ISmaccComponent, TSmaccObjectType>::value,
314  "Only are accepted smacc types as subscribers for smacc signals");
315 
316  typedef decltype(callback) ft;
317  Bind<boost::function_types::function_arity<ft>::value> binder;
318  boost::signals2::connection connection = binder.bindaux(signal, callback, object);
319 
320  // long life-time objects
321  if (std::is_base_of<ISmaccComponent, TSmaccObjectType>::value ||
322  std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
323  std::is_base_of<ISmaccOrthogonal, TSmaccObjectType>::value ||
324  std::is_base_of<ISmaccStateMachine, TSmaccObjectType>::value)
325  {
326  }
327  else if (std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
328  std::is_base_of<StateReactor, TSmaccObjectType>::value ||
329  std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value)
330  {
331  ROS_INFO("[StateMachine] life-time constrained smacc signal subscription created. Subscriber is %s",
332  demangledTypeName<TSmaccObjectType>().c_str());
333  stateCallbackConnections.push_back(connection);
334  }
335  else // state life-time objects
336  {
337  ROS_WARN("[StateMachine] connecting signal to an unknown object with life-time unknown behavior. It might provoke "
338  "an exception if the object is destroyed during the execution.");
339  }
340 
341  return connection;
342  }
