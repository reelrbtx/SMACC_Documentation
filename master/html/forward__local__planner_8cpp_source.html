137 Eigen::Vector3f ForwardLocalPlanner::computeNewPositions(const Eigen::Vector3f &pos, const Eigen::Vector3f &vel, double dt)
138 {
139  Eigen::Vector3f new_pos = Eigen::Vector3f::Zero();
140  new_pos[0] = pos[0] + (vel[0] * cos(pos[2]) + vel[1] * cos(M_PI_2 + pos[2])) * dt;
141  new_pos[1] = pos[1] + (vel[0] * sin(pos[2]) + vel[1] * sin(M_PI_2 + pos[2])) * dt;
142  new_pos[2] = pos[2] + vel[2] * dt;
143  return new_pos;
144 }

215 void clamp(geometry_msgs::Twist &cmd_vel, double max_linear_x_speed_, double max_angular_z_speed_)
216 {
217  if (max_angular_z_speed_ == 0 || max_linear_x_speed_ == 0)
218  return;
219 
220  if (cmd_vel.angular.z == 0)
221  {
222  cmd_vel.linear.x = max_linear_x_speed_;
223  }
224  else
225  {
226  double kurvature = cmd_vel.linear.x / cmd_vel.angular.z;
227 
228  double linearAuthority = fabs(cmd_vel.linear.x / max_linear_x_speed_);
229  double angularAuthority = fabs(cmd_vel.angular.z / max_angular_z_speed_);
230  if (linearAuthority < angularAuthority)
231  {
232  // lets go to maximum linear speed
233  cmd_vel.linear.x = max_linear_x_speed_;
234  cmd_vel.angular.z = kurvature / max_linear_x_speed_;
235  ROS_WARN_STREAM("k=" << kurvature << "lets go to maximum linear capacity: " << cmd_vel);
236  }
237  else
238  {
239  // lets go with maximum angular speed
240  cmd_vel.angular.x = max_angular_z_speed_;
241  cmd_vel.linear.x = kurvature * max_angular_z_speed_;
242  ROS_WARN_STREAM("lets go to maximum angular capacity: " << cmd_vel);
243  }
244  }

254  goalReached_ = false;
255  //ROS_DEBUG("LOCAL PLANNER LOOP");
256 
257  tf::Stamped<tf::Pose> tfpose = optionalRobotPose(costmapRos_);
258 
259  tf::Quaternion q = tfpose.getRotation();
260 
261  bool ok = false;
262  while (!ok)
263  {
264  // iterate the point from the current position and ahead until reaching a new goal point in the path
265  for (; !ok && currentPoseIndex_ < plan_.size(); currentPoseIndex_++)
266  {
267  auto &pose = plan_[currentPoseIndex_];
268  const geometry_msgs::Point &p = pose.pose.position;
269  tf::Quaternion q;
270  tf::quaternionMsgToTF(pose.pose.orientation, q);
271 
272  // take error from the current position to the path point
273  double dx = p.x - tfpose.getOrigin().x();
274  double dy = p.y - tfpose.getOrigin().y();
275  double dist = sqrt(dx * dx + dy * dy);
276 
277  double pangle = tf::getYaw(q);
278  double angle = tf::getYaw(tfpose.getRotation());
279  double angular_error = angles::shortest_angular_distance(pangle, angle);
280 
281  if (dist >= carrot_distance_ || angular_error > 0.1)
282  {
283  // the target pose is enough different to be defined as a target
284  ok = true;
285  //ROS_INFO("forward: %lf", 100.0 * currentPoseIndex_ / (double)plan_.size());
286  }
287  }
288 
289  if (currentPoseIndex_ >= plan_.size())
290  {
291  // even the latest point is quite similar, then take the last since it is the final goal
292  cmd_vel.linear.x = 0;
293  cmd_vel.angular.z = 0;
294  //ROS_INFO("End Local planner");
295  ok = true;
296  currentPoseIndex_ = plan_.size() - 1;
297  //return true;
298  }
299  }
300 
301  //ROS_INFO("pose control algorithm");
302 
303  const geometry_msgs::PoseStamped &finalgoalpose = plan_.back();
304  const geometry_msgs::PoseStamped &goalpose = plan_[currentPoseIndex_];
305  const geometry_msgs::Point &goalposition = goalpose.pose.position;
306 
307  tf::Quaternion goalQ;
308  tf::quaternionMsgToTF(goalpose.pose.orientation, goalQ);
309  //ROS_INFO_STREAM("Plan goal quaternion at "<< goalpose.pose.orientation);
310 
311  //goal orientation (global frame)
312  double betta = tf::getYaw(goalpose.pose.orientation) + betta_offset_;
313 
314  double dx = goalposition.x - tfpose.getOrigin().x();
315  double dy = goalposition.y - tfpose.getOrigin().y();
316 
317  //distance error to the targetpoint
318  double rho_error = sqrt(dx * dx + dy * dy);
319 
320  //current angle
321  double theta = tf::getYaw(q);
322  double alpha = atan2(dy, dx);
323  alpha = alpha + alpha_offset_;
324 
325  double alpha_error = angles::shortest_angular_distance(alpha, theta);
326  double betta_error = angles::shortest_angular_distance(betta, theta);
327 
328  double vetta; // = k_rho_ * rho_error;
329  double gamma; //= k_alpha_ * alpha_error + k_betta_ * betta_error;
330 
331  if (rho_error > xy_goal_tolerance_)
332  {
333  vetta = k_rho_ * rho_error;
334  gamma = k_alpha_ * alpha_error;
335  }
336  else if (fabs(betta_error) >= 0.01)
337  {
338  vetta = 0;
339  gamma = k_betta_ * betta_error;
340  }
341  else
342  {
343  vetta = 0;
344  gamma = 0;
345  goalReached_ = true;
346  }
347 
348  // linear speed clamp
349  if (vetta > max_linear_x_speed_)
350  {
351  vetta = max_linear_x_speed_;
352  }
353  else if (vetta < -max_linear_x_speed_)
354  {
355  vetta = -max_linear_x_speed_;
356  }
357 
358  // angular speed clamp
359  if (gamma > max_angular_z_speed_)
360  {
361  gamma = max_angular_z_speed_;
362  }
363  else if (gamma < -max_angular_z_speed_)
364  {
365  gamma = -max_angular_z_speed_;
366  }
367

374 
375  publishGoalMarker(goalposition.x, goalposition.y, betta);
376 
377  ROS_DEBUG_STREAM("Forward local planner," << std::endl
378  << " theta: " << theta << std::endl

380  << " err_x: " << dx << std::endl

391  //}
