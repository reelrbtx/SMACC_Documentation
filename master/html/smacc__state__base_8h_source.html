299  {
300  this->getStateMachine().notifyOnStateEntryStart(static_cast<MostDerived *>(this));
301 
302  std::string classname = smacc::utils::cleanShortTypeName(typeid(MostDerived));
303 
304  // TODO: make this static to build the parameter tree at startup

307  ROS_DEBUG("nodehandle namespace: %s", nh.getNamespace().c_str());

309  this->setParam("created", true);
310 
311  // before dynamic runtimeConfigure, we execute the staticConfigure behavior configurations
312  {
313  ROS_INFO("-- STATIC STATE DESCRIPTION --");
314 
315  for (const auto &stateReactorsVector : SmaccStateInfo::staticBehaviorInfo)
316  {
317  ROS_DEBUG_STREAM(" - state info: " << demangleSymbol(stateReactorsVector.first->name()));
318  for (auto &bhinfo : stateReactorsVector.second)
319  {
320  ROS_DEBUG_STREAM(" - client behavior: " << demangleSymbol(bhinfo.behaviorType->name()));
321  }
322  }
323 
324  const std::type_info *tindex = &(typeid(MostDerived));
325  auto &staticDefinedBehaviors = SmaccStateInfo::staticBehaviorInfo[tindex];
326  auto &staticDefinedStateReactors = SmaccStateInfo::stateReactorsInfo[tindex];
327 
328  for (auto &bhinfo : staticDefinedBehaviors)
329  {
330  ROS_INFO_STREAM("- Creating static client behavior: " << demangleSymbol(bhinfo.behaviorType->name()));
331  bhinfo.factoryFunction(this);
332  }
333 
334  for (auto &sr : staticDefinedStateReactors)
335  {
336  ROS_INFO_STREAM("- Creating static state reactor: " << demangleSymbol(sr->stateReactorType->name()));
337  sr->factoryFunction(this);
338  }
339 
340  ROS_INFO("---- END STATIC DESCRIPTION");
341  }
342 
343  ROS_INFO("State runtime configuration");
344 
345  // first we runtime configure the state, where we create client behaviors
346  static_cast<MostDerived *>(this)->runtimeConfigure();
347 
348  // second the orthogonals are internally configured
349  this->getStateMachine().notifyOnRuntimeConfigured(static_cast<MostDerived *>(this));
350 
351  // finally we go to the derived state onEntry Function
352  static_cast<MostDerived *>(this)->onEntry();
