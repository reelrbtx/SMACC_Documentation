134 void SignalDetector::pollOnce()
135 {
136  if (smaccStateMachine_ == nullptr)
137  {
138  ROS_DEBUG("[PollOnce] update but state machine is not yet set.");
139  return;
140  }
141 
142  try
143  {
144  smaccStateMachine_->lockStateMachine("update behaviors");
145 
146  ROS_DEBUG_STREAM("updatable clients: " << this->updatableClients_.size());
147 
148  if (this->updatableClients_.size())
149  {
150  for (auto *updatableClient : this->updatableClients_)
151  {
152  ROS_DEBUG_STREAM("[PollOnce] update client call: " << demangleType(typeid(updatableClient)));
153  updatableClient->update();
154  }
155  }
156 
157  long currentStateIndex = smaccStateMachine_->getCurrentStateCounter();
158 
159  ROS_DEBUG_STREAM("[PollOnce] update behaviors. checking current state");
160 
161  if (smaccStateMachine_->getCurrentState() != nullptr)
162  {
163  ROS_DEBUG_STREAM("[PollOnce] current state: " << currentStateIndex);
164  ROS_DEBUG_STREAM("[PollOnce] last state: " << this->lastState_);
165 
166  if (currentStateIndex != 0)
167  {
168  if (currentStateIndex != this->lastState_)
169  {
170  ROS_DEBUG_STREAM("[PollOnce] detected new state, refreshing updatable client behavior table");
171  // we are in a new state, refresh the updatable client behaviors table
172  this->lastState_ = currentStateIndex;
173  this->findUpdatableBehaviors();
174  }
175 
176  ROS_DEBUG_STREAM("updatable client_behaviors: " << this->updatableClientBehaviors_.size());
177  for (auto *updatableBehavior : this->updatableClientBehaviors_)
178  {
179  ROS_DEBUG_STREAM("pollOnce update client behavior call: " << demangleType(typeid(*updatableBehavior)));
180  updatableBehavior->update();
181  }
182  }
183  }
184  }
185  catch (...)
186  {
187  ROS_ERROR("Exception during Signal Detector update loop.");
188  }
189  smaccStateMachine_->unlockStateMachine("update behaviors");
190 }
