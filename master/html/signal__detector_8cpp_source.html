178  void SignalDetector::pollOnce()
179  {
180  // precondition: smaccStateMachine_ != nullptr
181 
182  try
183  {
184  smaccStateMachine_->lockStateMachine("update behaviors");
185 
186  this->findUpdatableClients();
187  ROS_DEBUG_STREAM("updatable clients: " << this->updatableClients_.size());
188 
189  if (this->updatableClients_.size())
190  {
191  for (auto *updatableClient : this->updatableClients_)
192  {
193  ROS_DEBUG_STREAM("[PollOnce] update client call: " << demangleType(typeid(updatableClient)));
194  updatableClient->executeUpdate();
195  }
196  }
197 
198  // STATE UPDATABLE ELEMENTS
199  if (this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::TRANSITIONING &&
200  this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::STATE_CONFIGURING &&
201  this->smaccStateMachine_->stateMachineCurrentAction != StateMachineInternalAction::STATE_EXITING)
202  {
203  // we do not update updatable elements during trasitioning or configuration of states
204  long currentStateIndex = smaccStateMachine_->getCurrentStateCounter();
205 
206  ROS_DEBUG_STREAM("[PollOnce] update behaviors. checking current state");
207 
208  auto currentState = smaccStateMachine_->getCurrentState();
209  if (currentState != nullptr)
210  {
211  ROS_DEBUG_STREAM("[PollOnce] current state: " << currentStateIndex);
212  ROS_DEBUG_STREAM("[PollOnce] last state: " << this->lastState_);
213 
214  if (currentStateIndex != 0)
215  {
216  if (currentStateIndex != this->lastState_)
217  {
218  ROS_DEBUG_STREAM("[PollOnce] detected new state, refreshing updatable client behavior table");
219  // we are in a new state, refresh the updatable client behaviors table
220  this->lastState_ = currentStateIndex;
221  this->findUpdatableStateElements(currentState);
222  }
223 
224  ROS_DEBUG_STREAM("updatable state elements: " << this->updatableStateElements_.size());
225  for (auto *udpatableStateElement : this->updatableStateElements_)
226  {
227  ROS_DEBUG_STREAM("pollOnce update client behavior call: " << demangleType(typeid(*udpatableStateElement)));
228  udpatableStateElement->executeUpdate();
229  }
230  }
231  }
232  }
233  }
234  catch (std::exception& ex)
235  {
236  ROS_ERROR("Exception during Signal Detector update loop. %s", ex.what());
237  }
238 
239  smaccStateMachine_->unlockStateMachine("update behaviors");
240  }

247  void SignalDetector::pollingLoop()
248  {
249  //ros::NodeHandle nh("~"); // use node name as root of the parameter server
250 
251  ros::NodeHandle _;
252  ros::Rate r0(20);
253  while (!initialized_)
254  {
255  r0.sleep();
256  }
257 
258  ros::NodeHandle nh(smacc::utils::cleanShortTypeName(typeid(*this->smaccStateMachine_)));
259 
260  if (!nh.getParam("signal_detector_loop_freq", this->loop_rate_hz))
261  {
262  ROS_WARN("Signal detector frequency (ros param ~signal_detector_loop_freq) was not set, using default frequency: %lf", this->loop_rate_hz);
263  }
264  else
265  {
266  ROS_WARN("Signal detector frequency (ros param ~signal_detector_loop_freq): %lf", this->loop_rate_hz);
267  }
268 
269  nh.setParam("signal_detector_loop_freq", this->loop_rate_hz);
270 
271  ROS_INFO_STREAM("[SignalDetector] loop rate hz:" << loop_rate_hz);
272 
273  ros::Rate r(loop_rate_hz);
274  while (ros::ok() && !end_)
275  {
276  ROS_INFO_STREAM_THROTTLE(10, "[SignalDetector] heartbeat");
277  pollOnce();
278  ros::spinOnce();
279  r.sleep();
