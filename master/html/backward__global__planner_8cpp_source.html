121 bool BackwardGlobalPlanner::createPureSpiningAndStragihtLineBackwardPath(const geometry_msgs::PoseStamped &start,
122  const geometry_msgs::PoseStamped &goal, std::vector<geometry_msgs::PoseStamped> &plan)
123 {
124  //tf::Stamped<tf::Pose> tfpose;
125  //ROS_WARN_NAMED("Backwards", "getting robot pose referencing costmap: %ld", (long)costmap_ros_);
126 
127  auto q = start.pose.orientation;
128 
129  double dx = start.pose.position.x - goal.pose.position.x;
130  double dy = start.pose.position.y - goal.pose.position.y;
131 
132  double lenght = sqrt(dx * dx + dy * dy);
133 
134  geometry_msgs::PoseStamped prevState;
135  if (lenght > skip_straight_motion_distance_)
136  {
137  // skip initial pure spinning and initial straight motion
138  //ROS_INFO("1 - heading to goal position pure spinning");
139  double heading_direction = atan2(dy, dx);
140  double startyaw = tf::getYaw(q);
141  double offset = angles::shortest_angular_distance(startyaw, heading_direction);
142  heading_direction = startyaw + offset;
143 
144  prevState = cl_move_base_z::makePureSpinningSubPlan(start, heading_direction, plan, puresSpinningRadStep_);
145  //ROS_INFO("2 - going forward keep orientation pure straight");
146 
147  prevState = cl_move_base_z::makePureStraightSubPlan(prevState, goal.pose.position, lenght, plan);
148  }
149  else
150  {
151  prevState = start;
152  }
153 }

160 bool BackwardGlobalPlanner::createDefaultBackwardPath(const geometry_msgs::PoseStamped &start,
161  const geometry_msgs::PoseStamped &goal, std::vector<geometry_msgs::PoseStamped> &plan)
162 {
163  auto q = start.pose.orientation;
164 
165  geometry_msgs::PoseStamped pose;
166  pose = start;
167 
168  plan.push_back(pose);
169 
170  //ROS_WARN_NAMED("Backwards", "Iterating in last forward cord path");
171  int i = lastForwardPathMsg_.poses.size();
172  double mindist = std::numeric_limits<double>::max();
173  int mindistindex = -1;
174 
175  geometry_msgs::Pose goalProjected;
176 
177  for (auto &p : lastForwardPathMsg_.poses | boost::adaptors::reversed)
178  {
179  pose = p;
180  pose.header.frame_id = costmap_ros_->getGlobalFrameID();
181  pose.header.stamp = ros::Time::now();
182 
183  double dx = pose.pose.position.x - goal.pose.position.x;
184  double dy = pose.pose.position.y - goal.pose.position.y;
185 
186  double dist = sqrt(dx * dx + dy * dy);
187  if (dist <= mindist)
188  {
189  mindistindex = i;
190  mindist = dist;
191  goalProjected = pose.pose;
192  }
193 
194  i--;
195  }
196 
197  if (mindistindex != -1)
198  {
199  for (int i = lastForwardPathMsg_.poses.size() - 1; i >= mindistindex; i--)
200  {
201  auto &pose = lastForwardPathMsg_.poses[i];
202  plan.push_back(pose);
203  }
204  }
205  else
206  {
207  ROS_WARN_STREAM("Creating the backwards plan, it is not found any close trajectory point. Last forward path plan message size: " << lastForwardPathMsg_.poses.size());
208  }
209 }

216 bool BackwardGlobalPlanner::makePlan(const geometry_msgs::PoseStamped &start,
217  const geometry_msgs::PoseStamped &goal, std::vector<geometry_msgs::PoseStamped> &plan)
218 {
219  //ROS_WARN_NAMED("Backwards", "Backwards global planner: Generating global plan ");

222  plan.clear();
223

225  //this->createPureSpiningAndStragihtLineBackwardPath(start, goal, plan);

234  //ROS_WARN_STREAM( "MAKE PLAN INVOKED, plan size:"<< plan.size());
235  publishGoalMarker(goal.pose, 1.0, 0, 1.0);
236 
237  if (plan.size() <= 1)
238  {
239  // if the backward plan is too much small, we create one artificial path
240  // with two poses (the current pose) so that we do not reject the plan request
241  plan.clear();
242  plan.push_back(start);
243  plan.push_back(start);
244  ROS_INFO("Artificial backward plan on current pose.");
245  }
246 
247  nav_msgs::Path planMsg;
248  planMsg.poses = plan;
249  planMsg.header.frame_id = "/odom";
250 
251  // check plan rejection
252  bool acceptedGlobalPlan = true;
253 
254  // static const unsigned char NO_INFORMATION = 255;
255  // static const unsigned char LETHAL_OBSTACLE = 254;
256  // static const unsigned char INSCRIBED_INFLATED_OBSTACLE = 253;
257  // static const unsigned char FREE_SPACE = 0;
258 
259  costmap_2d::Costmap2D *costmap2d = this->costmap_ros_->getCostmap();
260  for (auto &p : plan)
261  {
262  unsigned int mx, my;
263  costmap2d->worldToMap(p.pose.position.x, p.pose.position.y, mx, my);
264  auto cost = costmap2d->getCost(mx, my);
265 
266  if (cost >= costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
267  {
268  acceptedGlobalPlan = false;
269  break;
270  }
271  }
272 
273  planPub_.publish(planMsg);
274 
275  // this was previously set to size() <= 1, but a plan with a single point is also a valid plan (the goal)
276  return true;

298 bool BackwardGlobalPlanner::commandServiceCall(::backward_global_planner::command::Request &req, ::backward_global_planner::command::Response &res)
299 {
300  //ROS_INFO_NAMED("Backwards", "BackwardGlobalplanner SERVICE CALL");
301  std::string msg = req.cmd.data.c_str();
302 
303  std::vector<std::string> fields; // Create a vector of strings, called "fields"
304  boost::split(fields, msg, boost::algorithm::is_any_of(" "));
305 
306  //ROS_INFO_NAMED("Backwards", "backward planner SERVICE REQUEST");
307 
308  if (fields.size() == 0)
309  {
310  res.success.data = false;
311  return false;
312  }
313 
314  std::string cmd = fields[0];
315  bool error = false;
316  if (cmd == "savepath")
317  {
318  //ROS_INFO_NAMED("Backwards","SAVE PATH COMMAND");
319  if (fields.size() > 1)
320  {
321  std::vector<std::string> tail(fields.begin() + 1, fields.end());
322  std::string filename = boost::algorithm::join(tail, " ");
323 
324  std::ofstream os;
325  os.open(filename);
326  os << lastForwardPathMsg_;
327  os.close();
328 
329  //ROS_INFO_STREAM("serialized path: " << lastForwardPathMsg_);
330  }
331  else
332  {
333  error = true;
334  }
335  }
336  else if (cmd == "loadpath")
337  {
338  //ROS_INFO_NAMED("Backwards", "LOAD PATH COMMAND");
339  if (fields.size() > 1)
340  {
341  std::vector<std::string> tail(fields.begin() + 1, fields.end());
342  std::string filename = boost::algorithm::join(tail, " ");
343 
344  std::ifstream ifs(filename);
345  std::string alltext((std::istreambuf_iterator<char>(ifs)), std::istreambuf_iterator<char>());
346 
347  nav_msgs::Path p;
348  uint32_t serial_size = ros::serialization::serializationLength(p);
349  boost::shared_array<uint8_t> buffer(new uint8_t[serial_size]);
350  ros::serialization::IStream stream(buffer.get(), serial_size);
351  ros::serialization::deserialize(stream, p);
352 
353  //ROS_INFO_STREAM_NAMED("Backwards", "serialized path: " << p);
354  lastForwardPathMsg_ = p;
355  }
356  else
357  {
358  error = true;
359  }
360  }
361  else
362  {
363  res.success.data = false;
364  return false;
365  }
366 
367  res.success.data = error;
368  return true;
