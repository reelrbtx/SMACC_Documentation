117 bool BackwardGlobalPlanner::createDefaultBackwardPath(const geometry_msgs::PoseStamped &start,
118  const geometry_msgs::PoseStamped &goal, std::vector<geometry_msgs::PoseStamped> &plan)
119 {
120  auto q = start.pose.orientation;
121 
122  geometry_msgs::PoseStamped pose;
123  pose = start;
124 
125  double dx = start.pose.position.x - goal.pose.position.x;
126  double dy = start.pose.position.y - goal.pose.position.y;
127 
128  double lenght = sqrt(dx * dx + dy * dy);
129 
130  geometry_msgs::PoseStamped prevState;
131  if (lenght > skip_straight_motion_distance_)
132  {
133  // skip initial pure spinning and initial straight motion
134  //ROS_INFO("1 - heading to goal position pure spinning");
135  double heading_direction = atan2(dy, dx);
136  double startyaw = tf::getYaw(q);
137  double offset = angles::shortest_angular_distance(startyaw, heading_direction);
138  heading_direction = startyaw + offset;
139 
140  prevState = cl_move_base_z::makePureSpinningSubPlan(start, heading_direction, plan, puresSpinningRadStep_);
141  //ROS_INFO("2 - going forward keep orientation pure straight");
142 
143  prevState = cl_move_base_z::makePureStraightSubPlan(prevState, goal.pose.position, lenght, plan);
144  }
145  else
146  {
147  prevState = start;
148  }
149 
150  ROS_WARN_STREAM( "[BackwardGlobalPlanner ] backward global plan size: " <<plan.size());
151 }
