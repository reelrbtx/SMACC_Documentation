671  bool BackwardLocalPlanner::resamplePrecisePlan()
672  {
673  // this algorithm is really important to have a precise carrot (linear or angular)
674  // and not being considered as a divergence from the path
675 
676  ROS_DEBUG("[BackwardLocalPlanner] resample precise");
677  if(backwardsPlanPath_.size() <=1)
678  {
679  ROS_DEBUG_STREAM("[BackwardLocalPlanner] resample precise skipping, size: " << backwardsPlanPath_.size());
680  return false;
681  }
682 
683  int counter = 0;
684  double maxallowedAngularError = 0.45 * this->carrot_angular_distance_; // nyquist
685  double maxallowedLinearError = 0.45 * this->carrot_distance_; // nyquist
686 
687  for (int i = 0; i < backwardsPlanPath_.size() - 1; i++)
688  {
689  ROS_DEBUG_STREAM("[BackwardLocalPlanner] resample precise, check: " << i);
690  auto &currpose = backwardsPlanPath_[i];
691  auto &nextpose = backwardsPlanPath_[i + 1];
692 
693  tf::Quaternion qCurrent, qNext;
694  tf::quaternionMsgToTF(currpose.pose.orientation, qCurrent);
695  tf::quaternionMsgToTF(nextpose.pose.orientation, qNext);
696 
697  double dx = nextpose.pose.position.x - currpose.pose.position.x;
698  double dy = nextpose.pose.position.y - currpose.pose.position.y;
699  double dist = sqrt(dx * dx + dy * dy);
700 
701  bool resample = false;
702  if (dist > maxallowedLinearError)
703  {
704  ROS_DEBUG_STREAM("[BackwardLocalPlanner] resampling point, linear distance:" << dist << "(" << maxallowedLinearError << ")" << i);
705  resample = true;
706  }
707  else
708  {
709  double currentAngle = tf::getYaw(qCurrent);
710  double nextAngle = tf::getYaw(qNext);
711 
712  double angularError = fabs(angles::shortest_angular_distance(currentAngle, nextAngle));
713  if (angularError > maxallowedAngularError)
714  {
715  resample = true;
716  ROS_DEBUG_STREAM("[BackwardLocalPlanner] resampling point, angular distance:" << angularError << "(" << maxallowedAngularError << ")" << i);
717  }
718  }
719 
720  if (resample)
721  {
722  geometry_msgs::PoseStamped pintermediate;
723  auto duration = nextpose.header.stamp - currpose.header.stamp;
724  pintermediate.header.frame_id = currpose.header.frame_id;
725  pintermediate.header.stamp = currpose.header.stamp + duration *0.5;
726 
727  pintermediate.pose.position.x = 0.5 * (currpose.pose.position.x + nextpose.pose.position.x);
728  pintermediate.pose.position.y = 0.5 * (currpose.pose.position.y + nextpose.pose.position.y);
729  pintermediate.pose.position.z = 0.5 * (currpose.pose.position.z + nextpose.pose.position.z);
730  tf::Quaternion intermediateQuat = tf::slerp(qCurrent, qNext, 0.5);
731  tf::quaternionTFToMsg(intermediateQuat, pintermediate.pose.orientation);
732 
733  this->backwardsPlanPath_.insert(this->backwardsPlanPath_.begin() + i + 1, pintermediate);
734 
735  // retry this point
736  i--;
737  counter++;
738  }
739  }
740 
741  ROS_DEBUG_STREAM("[BackwardLocalPlanner] " << counter << " new inserted poses during precise resmapling.");
742  return true;
743  }
