200 void Robot::read(std::function<bool(const RobotState&)> read_callback) {
201  std::unique_lock<std::mutex> l(control_mutex_, std::try_to_lock);
202  if (!l.owns_lock()) {
203  throw InvalidOperationException(
204  "libfranka robot: Cannot perform this operation while another control or read operation "
205  "is running.");
206  }
207 
208  while (true) {
209  RobotState robot_state = impl_->update(nullptr, nullptr);
210  if (!read_callback(robot_state)) {
211  break;
212  }
213  }
214 }
