13 Eigen::Vector3d limitRate(double max_velocity,
14  double max_acceleration,
15  double max_jerk,
16  const Eigen::Vector3d& commanded_velocity,
17  const Eigen::Vector3d& last_commanded_velocity,
18  const Eigen::Vector3d& last_commanded_acceleration) {
19  // Differentiate to get jerk
20  Eigen::Vector3d commanded_jerk =
21  (((commanded_velocity - last_commanded_velocity) / kDeltaT) - last_commanded_acceleration) /
22  kDeltaT;
23 
24  // Limit jerk and integrate to get desired acceleration
25  Eigen::Vector3d commanded_acceleration = last_commanded_acceleration;
26  if (commanded_jerk.norm() > kNormEps) {
27  commanded_acceleration += (commanded_jerk / commanded_jerk.norm()) *
28  std::max(std::min(commanded_jerk.norm(), max_jerk), -max_jerk) *
29  kDeltaT;
30  }
31 
32  // Compute Euclidean distance to the max velocity vector that would be reached starting from
33  // last_commanded_velocity with the direction of the desired acceleration
34  Eigen::Vector3d unit_commanded_acceleration =
35  commanded_acceleration / commanded_acceleration.norm();
36  double dot_product = unit_commanded_acceleration.transpose() * last_commanded_velocity;
37  double distance_to_max_velocity =
38  -dot_product + std::sqrt(pow(dot_product, 2.0) - last_commanded_velocity.squaredNorm() +
39  pow(max_velocity, 2.0));
40 
41  // Compute safe acceleration limits
42  double safe_max_acceleration =
43  std::min((max_jerk / max_acceleration) * distance_to_max_velocity, max_acceleration);
44 
45  // Limit acceleration and integrate to get desired velocities
46  Eigen::Vector3d limited_commanded_velocity = last_commanded_velocity;
47 
48  if (commanded_acceleration.norm() > kNormEps) {
49  limited_commanded_velocity += unit_commanded_acceleration *
50  std::min(commanded_acceleration.norm(), safe_max_acceleration) *
51  kDeltaT;
52  }
53 
54  return limited_commanded_velocity;
55 }

59 std::array<double, 7> limitRate(const std::array<double, 7>& max_derivatives,
60  const std::array<double, 7>& commanded_values,
61  const std::array<double, 7>& last_commanded_values) {
62  if (!std::all_of(commanded_values.begin(), commanded_values.end(),
63  [](double d) { return std::isfinite(d); })) {
64  throw std::invalid_argument("Commanding value is infinite or NaN.");
65  }
66  std::array<double, 7> limited_values{};
67  for (size_t i = 0; i < 7; i++) {
68  double commanded_derivative = (commanded_values[i] - last_commanded_values[i]) / kDeltaT;
69  limited_values[i] =
70  last_commanded_values[i] +
71  std::max(std::min(commanded_derivative, max_derivatives[i]), -max_derivatives[i]) * kDeltaT;
72  }
73  return limited_values;
74 }

76 double limitRate(double max_velocity,
77  double max_acceleration,
78  double max_jerk,
79  double commanded_velocity,
80  double last_commanded_velocity,
81  double last_commanded_acceleration) {
82  if (!std::isfinite(commanded_velocity)) {
83  throw std::invalid_argument("commanded_velocity is infinite or NaN.");
84  }
85  // Differentiate to get jerk
86  double commanded_jerk =
87  (((commanded_velocity - last_commanded_velocity) / kDeltaT) - last_commanded_acceleration) /
88  kDeltaT;
89 
90  // Limit jerk and integrate to get acceleration
91  double commanded_acceleration = last_commanded_acceleration +
92  std::max(std::min(commanded_jerk, max_jerk), -max_jerk) * kDeltaT;
93 
94  // Compute acceleration limits
95  double safe_max_acceleration = std::min(
96  (max_jerk / max_acceleration) * (max_velocity - last_commanded_velocity), max_acceleration);
97  double safe_min_acceleration = std::max(
98  (max_jerk / max_acceleration) * (-max_velocity - last_commanded_velocity), -max_acceleration);
99 
100  // Limit acceleration and integrate to get desired velocities
101  return last_commanded_velocity +
102  std::max(std::min(commanded_acceleration, safe_max_acceleration), safe_min_acceleration) *
103  kDeltaT;
104 }

190 std::array<double, 16> limitRate(
191  double max_translational_velocity,
192  double max_translational_acceleration,
193  double max_translational_jerk,
194  double max_rotational_velocity,
195  double max_rotational_acceleration,
196  double max_rotational_jerk,
197  const std::array<double, 16>& O_T_EE_c, // NOLINT(readability-identifier-naming)
198  const std::array<double, 16>& last_O_T_EE_c, // NOLINT(readability-identifier-naming)
199  const std::array<double, 6>& last_O_dP_EE_c, // NOLINT(readability-identifier-naming)
200  const std::array<double, 6>& last_O_ddP_EE_c) { // NOLINT(readability-identifier-naming)
201  if (!std::all_of(O_T_EE_c.begin(), O_T_EE_c.end(), [](double d) { return std::isfinite(d); })) {
202  throw std::invalid_argument("O_T_EE_c is infinite or NaN.");
203  }
204  if (!isHomogeneousTransformation(O_T_EE_c)) {
205  throw std::invalid_argument(
206  "O_T_EE_c is invalid transformation matrix. Has to be column major!");
207  }
208  Eigen::Matrix<double, 6, 1> dx;
209  Eigen::Affine3d commanded_pose(Eigen::Matrix4d::Map(O_T_EE_c.data()));
210  Eigen::Affine3d limited_commanded_pose = Eigen::Affine3d::Identity();
211  Eigen::Affine3d last_commanded_pose(Eigen::Matrix4d::Map(last_O_T_EE_c.data()));
212 
213  // Compute translational velocity
214  dx.head(3) << (commanded_pose.translation() - last_commanded_pose.translation()) / kDeltaT;
215 
216  // Compute rotational velocity
217  Eigen::AngleAxisd rot_difference(commanded_pose.linear() *
218  last_commanded_pose.linear().transpose());
219  dx.tail(3) << rot_difference.axis() * rot_difference.angle() / kDeltaT;
220 
221  // Limit the rate of the twist
222  std::array<double, 6> commanded_O_dP_EE_c{}; // NOLINT(readability-identifier-naming)
223  Eigen::Map<Eigen::Matrix<double, 6, 1>>(&commanded_O_dP_EE_c[0], 6, 1) = dx;
224  commanded_O_dP_EE_c =
225  limitRate(max_translational_velocity, max_translational_acceleration, max_translational_jerk,
226  kFactorCartesianRotationPoseInterface * max_rotational_velocity,
227  kFactorCartesianRotationPoseInterface * max_rotational_acceleration,
228  kFactorCartesianRotationPoseInterface * max_rotational_jerk, commanded_O_dP_EE_c,
229  last_O_dP_EE_c, last_O_ddP_EE_c);
230  dx = Eigen::Matrix<double, 6, 1>(commanded_O_dP_EE_c.data());
231 
232  // Integrate limited twist
233  limited_commanded_pose.translation() << last_commanded_pose.translation() + dx.head(3) * kDeltaT;
234  limited_commanded_pose.linear() << last_commanded_pose.linear();
235  if (dx.tail(3).norm() > kNormEps) {
236  Eigen::Matrix3d omega_skew;
237  Eigen::Vector3d w_norm(dx.tail(3) / dx.tail(3).norm());
238  double theta = kDeltaT * dx.tail(3).norm();
239  omega_skew << 0, -w_norm(2), w_norm(1), w_norm(2), 0, -w_norm(0), -w_norm(1), w_norm(0), 0;
240  // NOLINTNEXTLINE(readability-identifier-naming)
241  Eigen::Matrix3d R = Eigen::Matrix3d::Identity() + sin(theta) * omega_skew +
242  (1.0 - cos(theta)) * (omega_skew * omega_skew);
243  limited_commanded_pose.linear() << R * last_commanded_pose.linear();
244  }
245 
246  std::array<double, 16> limited_values{};
247  Eigen::Map<Eigen::Matrix4d>(&limited_values[0], 4, 4) = limited_commanded_pose.matrix();
248  return limited_values;
249 }
