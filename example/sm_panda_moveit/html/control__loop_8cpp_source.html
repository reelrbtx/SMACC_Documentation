103 ControlLoop<T>::ControlLoop(RobotControl& robot,
104  ControllerMode controller_mode,
105  MotionGeneratorCallback motion_callback,
106  bool limit_rate,
107  double cutoff_frequency)
108  : ControlLoop(robot, std::move(motion_callback), {}, limit_rate, cutoff_frequency) {
109  if (!motion_callback_) {
110  throw std::invalid_argument("libfranka: Invalid motion callback given.");
111  }
112  research_interface::robot::Move::ControllerMode mode;
113  switch (controller_mode) {

122  }
123  motion_id_ = robot.startMotion(mode, MotionGeneratorTraits<T>::kMotionGeneratorMode,
124  kDefaultDeviation, kDefaultDeviation);

127 template <typename T>
128 void ControlLoop<T>::operator()() try {
129  RobotState robot_state = robot_.update(nullptr, nullptr);
130  robot_.throwOnMotionError(robot_state, motion_id_);
131 
132  Duration previous_time = robot_state.time;
133 
134  research_interface::robot::MotionGeneratorCommand motion_command{};
135  if (control_callback_) {
136  research_interface::robot::ControllerCommand control_command{};
137  while (spinMotion(robot_state, robot_state.time - previous_time, &motion_command) &&
138  spinControl(robot_state, robot_state.time - previous_time, &control_command)) {
139  previous_time = robot_state.time;
140  robot_state = robot_.update(&motion_command, &control_command);
141  robot_.throwOnMotionError(robot_state, motion_id_);
142  }
143  robot_.finishMotion(motion_id_, &motion_command, &control_command);
144  } else {
145  while (spinMotion(robot_state, robot_state.time - previous_time, &motion_command)) {
146  previous_time = robot_state.time;
147  robot_state = robot_.update(&motion_command, nullptr);
148  robot_.throwOnMotionError(robot_state, motion_id_);
149  }
150  robot_.finishMotion(motion_id_, &motion_command, nullptr);
151  }
152 } catch (...) {
153  try {
154  robot_.cancelMotion(motion_id_);
155  } catch (...) {
156  }
157  throw;
158 }

160 template <typename T>
161 bool ControlLoop<T>::spinControl(const RobotState& robot_state,
162  franka::Duration time_step,
163  research_interface::robot::ControllerCommand* command) {
164  Torques control_output = control_callback_(robot_state, time_step);
165  if (cutoff_frequency_ < kMaxCutoffFrequency) {
166  for (size_t i = 0; i < 7; i++) {
167  control_output.tau_J[i] = lowpassFilter(kDeltaT, control_output.tau_J[i],
168  robot_state.tau_J_d[i], cutoff_frequency_);
169  }
170  }
171  if (limit_rate_) {
172  control_output.tau_J = limitRate(kMaxTorqueRate, control_output.tau_J, robot_state.tau_J_d);
173  }
174  command->tau_J_d = control_output.tau_J;
175  checkFinite(command->tau_J_d);
176  return !control_output.motion_finished;
177 }

226 template <>
227 void ControlLoop<CartesianPose>::convertMotion(
228  const CartesianPose& motion,
229  const RobotState& robot_state,
230  research_interface::robot::MotionGeneratorCommand* command) {
231  command->O_T_EE_c = motion.O_T_EE;
232  if (cutoff_frequency_ < kMaxCutoffFrequency) {
233  command->O_T_EE_c =
234  cartesianLowpassFilter(kDeltaT, command->O_T_EE_c, robot_state.O_T_EE_c, cutoff_frequency_);
235  }
236 
237  if (limit_rate_) {
238  command->O_T_EE_c = limitRate(
239  kMaxTranslationalVelocity, kMaxTranslationalAcceleration, kMaxTranslationalJerk,
240  kMaxRotationalVelocity, kMaxRotationalAcceleration, kMaxRotationalJerk, command->O_T_EE_c,
241  robot_state.O_T_EE_c, robot_state.O_dP_EE_c, robot_state.O_ddP_EE_c);
242  }
243  checkMatrix(command->O_T_EE_c);
244 
245  if (motion.hasElbow()) {
246  command->valid_elbow = true;
247  command->elbow_c = motion.elbow;
248  if (cutoff_frequency_ < kMaxCutoffFrequency) {
249  command->elbow_c[0] =
250  lowpassFilter(kDeltaT, command->elbow_c[0], robot_state.elbow_c[0], cutoff_frequency_);
251  }
252  if (limit_rate_) {
253  command->elbow_c[0] =
254  limitRate(kMaxElbowVelocity, kMaxElbowAcceleration, kMaxElbowJerk, command->elbow_c[0],
255  robot_state.elbow_c[0], robot_state.delbow_c[0], robot_state.ddelbow_c[0]);
256  }
257  checkElbow(command->elbow_c);
258  } else {
259  command->valid_elbow = false;
260  command->elbow_c = {};
261  }
