11 double lowpassFilter(double sample_time, double y, double y_last, double cutoff_frequency) {
12  if (sample_time < 0 || !std::isfinite(sample_time)) {
13  throw std::invalid_argument("lowpass-filter: sample_time is negative, infinite or NaN.");
14  }
15  if (cutoff_frequency <= 0 || !std::isfinite(cutoff_frequency)) {
16  throw std::invalid_argument(
17  "lowpass-filter: cutoff_frequency is zero, negative, infinite or NaN.");
18  }
19  if (!std::isfinite(y) || !std::isfinite(y_last)) {
20  throw std::invalid_argument(
21  "lowpass-filter: current or past input value of the signal to be filtered is infinite or "
22  "NaN.");
23  }
24  double gain = sample_time / (sample_time + (1.0 / (2.0 * M_PI * cutoff_frequency)));
25  return gain * y + (1 - gain) * y_last;
26 }

41  if (!std::isfinite(y[i]) || !std::isfinite(y_last[i])) {

48  Eigen::Affine3d transform_last(Eigen::Matrix4d::Map(y_last.data()));
