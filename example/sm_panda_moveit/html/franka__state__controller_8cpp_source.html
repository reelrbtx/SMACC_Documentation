138 bool FrankaStateController::init(hardware_interface::RobotHW* robot_hardware,
139  ros::NodeHandle& root_node_handle,
140  ros::NodeHandle& controller_node_handle) {
141  franka_state_interface_ = robot_hardware->get<franka_hw::FrankaStateInterface>();
142  if (franka_state_interface_ == nullptr) {
143  ROS_ERROR("FrankaStateController: Could not get Franka state interface from hardware");
144  return false;
145  }
146  if (!controller_node_handle.getParam("arm_id", arm_id_)) {
147  ROS_ERROR("FrankaStateController: Could not get parameter arm_id");
148  return false;
149  }
150  double publish_rate(30.0);
151  if (!controller_node_handle.getParam("publish_rate", publish_rate)) {
152  ROS_INFO_STREAM("FrankaStateController: Did not find publish_rate. Using default "
153  << publish_rate << " [Hz].");
154  }
155  trigger_publish_ = franka_hw::TriggerRate(publish_rate);
156 
157  if (!controller_node_handle.getParam("joint_names", joint_names_) ||
158  joint_names_.size() != robot_state_.q.size()) {
159  ROS_ERROR(
160  "FrankaStateController: Invalid or no joint_names parameters provided, aborting "
161  "controller init!");
162  return false;
163  }
164 
165  try {
166  franka_state_handle_ = std::make_unique<franka_hw::FrankaStateHandle>(
167  franka_state_interface_->getHandle(arm_id_ + "_robot"));
168  } catch (const hardware_interface::HardwareInterfaceException& ex) {
169  ROS_ERROR_STREAM("FrankaStateController: Exception getting franka state handle: " << ex.what());
170  return false;
171  }
172 
173  publisher_transforms_.init(root_node_handle, "/tf", 1);
174  publisher_franka_states_.init(controller_node_handle, "franka_states", 1);
175  publisher_joint_states_.init(controller_node_handle, "joint_states", 1);
176  publisher_joint_states_desired_.init(controller_node_handle, "joint_states_desired", 1);
177  publisher_external_wrench_.init(controller_node_handle, "F_ext", 1);
178 
179  {
180  std::lock_guard<realtime_tools::RealtimePublisher<sensor_msgs::JointState> > lock(
181  publisher_joint_states_);
182  publisher_joint_states_.msg_.name.resize(joint_names_.size());
183  publisher_joint_states_.msg_.position.resize(robot_state_.q.size());
184  publisher_joint_states_.msg_.velocity.resize(robot_state_.dq.size());
185  publisher_joint_states_.msg_.effort.resize(robot_state_.tau_J.size());
186  }
187  {
188  std::lock_guard<realtime_tools::RealtimePublisher<sensor_msgs::JointState> > lock(
189  publisher_joint_states_desired_);
190  publisher_joint_states_desired_.msg_.name.resize(joint_names_.size());
191  publisher_joint_states_desired_.msg_.position.resize(robot_state_.q_d.size());
192  publisher_joint_states_desired_.msg_.velocity.resize(robot_state_.dq_d.size());
193  publisher_joint_states_desired_.msg_.effort.resize(robot_state_.tau_J_d.size());
194  }
195  {
196  std::lock_guard<realtime_tools::RealtimePublisher<tf2_msgs::TFMessage> > lock(
197  publisher_transforms_);
198  publisher_transforms_.msg_.transforms.resize(2);
199  tf::Quaternion quaternion(0.0, 0.0, 0.0, 1.0);
200  tf::Vector3 translation(0.0, 0.0, 0.05);
201  tf::Transform transform(quaternion, translation);
202  tf::StampedTransform stamped_transform(transform, ros::Time::now(), arm_id_ + "_link8",
203  arm_id_ + "_EE");
204  geometry_msgs::TransformStamped transform_message;
205  transformStampedTFToMsg(stamped_transform, transform_message);
206  publisher_transforms_.msg_.transforms[0] = transform_message;
207  translation = tf::Vector3(0.0, 0.0, 0.0);
208  transform = tf::Transform(quaternion, translation);
209  stamped_transform =
210  tf::StampedTransform(transform, ros::Time::now(), arm_id_ + "_EE", arm_id_ + "_K");
211  transformStampedTFToMsg(stamped_transform, transform_message);
212  publisher_transforms_.msg_.transforms[1] = transform_message;
213  }
214  {
215  std::lock_guard<realtime_tools::RealtimePublisher<geometry_msgs::WrenchStamped> > lock(
216  publisher_external_wrench_);
217  publisher_external_wrench_.msg_.header.frame_id = arm_id_ + "_K";
218  publisher_external_wrench_.msg_.wrench.force.x = 0.0;
219  publisher_external_wrench_.msg_.wrench.force.y = 0.0;
220  publisher_external_wrench_.msg_.wrench.force.z = 0.0;
221  publisher_external_wrench_.msg_.wrench.torque.x = 0.0;
222  publisher_external_wrench_.msg_.wrench.torque.y = 0.0;
223  publisher_external_wrench_.msg_.wrench.torque.z = 0.0;
224  }
225  return true;
226 }

239 void FrankaStateController::publishFrankaStates(const ros::Time& time) {
240  if (publisher_franka_states_.trylock()) {
241  static_assert(
242  sizeof(robot_state_.cartesian_collision) == sizeof(robot_state_.cartesian_contact),
243  "Robot state Cartesian members do not have same size");
244  static_assert(sizeof(robot_state_.cartesian_collision) == sizeof(robot_state_.K_F_ext_hat_K),
245  "Robot state Cartesian members do not have same size");
246  static_assert(sizeof(robot_state_.cartesian_collision) == sizeof(robot_state_.O_F_ext_hat_K),
247  "Robot state Cartesian members do not have same size");
248  for (size_t i = 0; i < robot_state_.cartesian_collision.size(); i++) {
249  publisher_franka_states_.msg_.cartesian_collision[i] = robot_state_.cartesian_collision[i];
250  publisher_franka_states_.msg_.cartesian_contact[i] = robot_state_.cartesian_contact[i];
251  publisher_franka_states_.msg_.K_F_ext_hat_K[i] = robot_state_.K_F_ext_hat_K[i];
252  publisher_franka_states_.msg_.O_F_ext_hat_K[i] = robot_state_.O_F_ext_hat_K[i];
253  }
254 
255  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.q_d),
256  "Robot state joint members do not have same size");
257  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.dq),
258  "Robot state joint members do not have same size");
259  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.dq_d),
260  "Robot state joint members do not have same size");
261  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.tau_J),
262  "Robot state joint members do not have same size");
263  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.dtau_J),
264  "Robot state joint members do not have same size");

266  "Robot state joint members do not have same size");
267  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.theta),
268  "Robot state joint members do not have same size");
269  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.dtheta),
270  "Robot state joint members do not have same size");
271  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.joint_collision),
272  "Robot state joint members do not have same size");
273  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.joint_contact),
274  "Robot state joint members do not have same size");
275  static_assert(sizeof(robot_state_.q) == sizeof(robot_state_.tau_ext_hat_filtered),
276  "Robot state joint members do not have same size");
277  for (size_t i = 0; i < robot_state_.q.size(); i++) {
278  publisher_franka_states_.msg_.q[i] = robot_state_.q[i];
279  publisher_franka_states_.msg_.q_d[i] = robot_state_.q_d[i];
280  publisher_franka_states_.msg_.dq[i] = robot_state_.dq[i];
281  publisher_franka_states_.msg_.dq_d[i] = robot_state_.dq_d[i];
282  publisher_franka_states_.msg_.tau_J[i] = robot_state_.tau_J[i];
283  publisher_franka_states_.msg_.dtau_J[i] = robot_state_.dtau_J[i];
284  publisher_franka_states_.msg_.tau_J_d[i] = robot_state_.tau_J_d[i];
285  publisher_franka_states_.msg_.theta[i] = robot_state_.theta[i];
286  publisher_franka_states_.msg_.dtheta[i] = robot_state_.dtheta[i];
287  publisher_franka_states_.msg_.joint_collision[i] = robot_state_.joint_collision[i];
288  publisher_franka_states_.msg_.joint_contact[i] = robot_state_.joint_contact[i];
289  publisher_franka_states_.msg_.tau_ext_hat_filtered[i] = robot_state_.tau_ext_hat_filtered[i];

291 
292  for (size_t i = 0; i < robot_state_.elbow.size(); i++) {

294  }
295 
296  for (size_t i = 0; i < robot_state_.elbow_d.size(); i++) {
297  publisher_franka_states_.msg_.elbow_d[i] = robot_state_.elbow_d[i];
298  }
299 
300  static_assert(sizeof(robot_state_.O_T_EE) == sizeof(robot_state_.F_T_EE),

302  static_assert(sizeof(robot_state_.O_T_EE) == sizeof(robot_state_.EE_T_K),
303  "Robot state transforms do not have same size");
304  static_assert(sizeof(robot_state_.O_T_EE) == sizeof(robot_state_.O_T_EE_d),
305  "Robot state transforms do not have same size");
306  for (size_t i = 0; i < robot_state_.O_T_EE.size(); i++) {
307  publisher_franka_states_.msg_.O_T_EE[i] = robot_state_.O_T_EE[i];

349
