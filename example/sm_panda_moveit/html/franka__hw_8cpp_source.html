159 void FrankaHW::control(
160  franka::Robot& robot,
161  const std::function<bool(const ros::Time&, const ros::Duration&)>& ros_callback) {
162  if (!controller_active_) {
163  return;
164  }
165 
166  franka::Duration last_time = robot_state_.time;
167 
168  run_function_(robot, [this, ros_callback, &last_time](const franka::RobotState& robot_state,
169  franka::Duration time_step) {
170  if (last_time != robot_state.time) {
171  last_time = robot_state.time;
172  return ros_callback(ros::Time::now(), ros::Duration(time_step.toSec()));
173  }
174  return true;
175  });

186 bool FrankaHW::checkForConflict(const std::list<hardware_interface::ControllerInfo>& info) const {
187  ResourceWithClaimsMap resource_map = getResourceMap(info);
188  // check for conflicts in single resources: no triple claims,
189  // for 2 claims it must be one torque and one non-torque claim
190  for (auto map_it = resource_map.begin(); map_it != resource_map.end(); map_it++) {
191  if (map_it->second.size() > 2) {
192  ROS_ERROR_STREAM("FrankaHW: Resource "
193  << map_it->first << " claimed with more than two interfaces. Conflict!");
194  return true;
195  }
196  uint8_t torque_claims = 0;
197  uint8_t other_claims = 0;
198  if (map_it->second.size() == 2) {
199  for (auto& claimed_by : map_it->second) {
200  if (claimed_by[2] == "hardware_interface::EffortJointInterface") {
201  torque_claims++;
202  } else {
203  other_claims++;
204  }
205  }
206  if (torque_claims != 1) {
207  ROS_ERROR_STREAM("FrankaHW: Resource "
208  << map_it->first
209  << " is claimed with two non-compatible interfaces. Conflict!");
210  return true;
211  }
212  }
213  }
214 
215  ArmClaimedMap arm_claim_map;
216  if (!getArmClaimedMap(resource_map, arm_claim_map)) {
217  ROS_ERROR_STREAM("FrankaHW: Unknown interface claimed. Conflict!");
218  return true;
219  }
220 
221  // check for conflicts between joint and cartesian level for each arm.
222  // Valid claims are torque claims on joint level in combination with either
223  // 7 non-torque claims on joint_level or one claim on cartesian level.
224  if (arm_claim_map.find(arm_id_) != arm_claim_map.end()) {
225  if ((arm_claim_map[arm_id_].cartesian_velocity_claims +
226  arm_claim_map[arm_id_].cartesian_pose_claims >
227  0 &&
228  arm_claim_map[arm_id_].joint_position_claims +
229  arm_claim_map[arm_id_].joint_velocity_claims >
230  0)) {
231  ROS_ERROR_STREAM("FrankaHW: Invalid claims on joint AND cartesian level on arm "
232  << arm_id_ << ". Conflict!");
233  return true;
234  }
235  if ((arm_claim_map[arm_id_].joint_position_claims > 0 &&
236  arm_claim_map[arm_id_].joint_position_claims != 7) ||
237  (arm_claim_map[arm_id_].joint_velocity_claims > 0 &&
238  arm_claim_map[arm_id_].joint_velocity_claims != 7) ||
239  (arm_claim_map[arm_id_].joint_torque_claims > 0 &&
240  arm_claim_map[arm_id_].joint_torque_claims != 7)) {
241  ROS_ERROR_STREAM("FrankaHW: Non-consistent claims on the joints of "
242  << arm_id_ << ". Not supported. Conflict!");
243  return true;
244  }
245  }
246  return false;
247 }

259 bool FrankaHW::prepareSwitch(const std::list<hardware_interface::ControllerInfo>& start_list,
260  const std::list<hardware_interface::ControllerInfo>& stop_list) {
261  ResourceWithClaimsMap start_resource_map = getResourceMap(start_list);
262  ArmClaimedMap start_arm_claim_map;
263  if (!getArmClaimedMap(start_resource_map, start_arm_claim_map)) {
264  ROS_ERROR("FrankaHW: Unknown interface claimed for starting!");
265  return false;
266  }
267  ControlMode start_control_mode = getControlMode(arm_id_, start_arm_claim_map);
268 
269  ResourceWithClaimsMap stop_resource_map = getResourceMap(stop_list);
270  ArmClaimedMap stop_arm_claim_map;
271  if (!getArmClaimedMap(stop_resource_map, stop_arm_claim_map)) {
272  ROS_ERROR("FrankaHW: Unknown interface claimed for stopping!");
273  return false;
274  }
275  ControlMode stop_control_mode = getControlMode(arm_id_, stop_arm_claim_map);
276 
277  ControlMode requested_control_mode = current_control_mode_;
278  requested_control_mode &= ~stop_control_mode;
279  requested_control_mode |= start_control_mode;
280 
281  auto limit_rate = get_limit_rate_();
282  auto cutoff_frequency = get_cutoff_frequency_();
283  auto internal_controller = get_internal_controller_();
284 
285  using std::placeholders::_1;
286  using std::placeholders::_2;
287 
288  switch (requested_control_mode) {
289  case ControlMode::None:
290  break;
291  case ControlMode::JointTorque:
292  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
293  robot.control(std::bind(&FrankaHW::controlCallback<franka::Torques>, this,
294  std::cref(effort_joint_command_), ros_callback, _1, _2),
295  limit_rate, cutoff_frequency);
296  };
297  break;
298  case ControlMode::JointPosition:
299  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
300  robot.control(std::bind(&FrankaHW::controlCallback<franka::JointPositions>, this,
301  std::cref(position_joint_command_), ros_callback, _1, _2),
302  internal_controller, limit_rate, cutoff_frequency);
303  };
304  break;
305  case ControlMode::JointVelocity:
306  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
307  robot.control(std::bind(&FrankaHW::controlCallback<franka::JointVelocities>, this,
308  std::cref(velocity_joint_command_), ros_callback, _1, _2),
309  internal_controller, limit_rate, cutoff_frequency);
310  };
311  break;
312  case ControlMode::CartesianPose:
313  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
314  robot.control(std::bind(&FrankaHW::controlCallback<franka::CartesianPose>, this,
315  std::cref(pose_cartesian_command_), ros_callback, _1, _2),
316  internal_controller, limit_rate, cutoff_frequency);
317  };
318  break;
319  case ControlMode::CartesianVelocity:
320  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
321  robot.control(std::bind(&FrankaHW::controlCallback<franka::CartesianVelocities>, this,
322  std::cref(velocity_cartesian_command_), ros_callback, _1, _2),
323  internal_controller, limit_rate, cutoff_frequency);
324  };
325  break;
326  case (ControlMode::JointTorque | ControlMode::JointPosition):
327  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
328  robot.control(std::bind(&FrankaHW::controlCallback<franka::Torques>, this,
329  std::cref(effort_joint_command_), ros_callback, _1, _2),
330  std::bind(&FrankaHW::controlCallback<franka::JointPositions>, this,
331  std::cref(position_joint_command_), ros_callback, _1, _2),
332  limit_rate, cutoff_frequency);
333  };
334  break;
335  case (ControlMode::JointTorque | ControlMode::JointVelocity):
336  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
337  robot.control(std::bind(&FrankaHW::controlCallback<franka::Torques>, this,
338  std::cref(effort_joint_command_), ros_callback, _1, _2),
339  std::bind(&FrankaHW::controlCallback<franka::JointVelocities>, this,
340  std::cref(velocity_joint_command_), ros_callback, _1, _2),
341  limit_rate, cutoff_frequency);
342  };
343  break;
344  case (ControlMode::JointTorque | ControlMode::CartesianPose):
345  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
346  robot.control(std::bind(&FrankaHW::controlCallback<franka::Torques>, this,
347  std::cref(effort_joint_command_), ros_callback, _1, _2),
348  std::bind(&FrankaHW::controlCallback<franka::CartesianPose>, this,
349  std::cref(pose_cartesian_command_), ros_callback, _1, _2),
350  limit_rate, cutoff_frequency);
351  };
352  break;
353  case (ControlMode::JointTorque | ControlMode::CartesianVelocity):
354  run_function_ = [=](franka::Robot& robot, Callback ros_callback) {
355  robot.control(std::bind(&FrankaHW::controlCallback<franka::Torques>, this,
356  std::cref(effort_joint_command_), ros_callback, _1, _2),
357  std::bind(&FrankaHW::controlCallback<franka::CartesianVelocities>, this,
358  std::cref(velocity_cartesian_command_), ros_callback, _1, _2),
359  limit_rate, cutoff_frequency);
360  };
361  break;
362  default:
363  ROS_WARN("FrankaHW: No valid control mode selected; cannot switch controllers.");
364  return false;
365  }
366 
367  if (current_control_mode_ != requested_control_mode) {
368  ROS_INFO_STREAM("FrankaHW: Prepared switching controllers to "
369  << requested_control_mode << " with parameters "
370  << "limit_rate=" << limit_rate << ", cutoff_frequency=" << cutoff_frequency
371  << ", internal_controller=" << internal_controller);
372  current_control_mode_ = requested_control_mode;
373 
374  controller_active_ = false;
375  }
376 
377  return true;
378 }
