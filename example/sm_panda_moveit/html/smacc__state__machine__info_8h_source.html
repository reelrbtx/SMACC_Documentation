160 {
161  ROS_INFO("State %s Walker transition: %s", sourceState->toShortName().c_str(), demangleSymbol(typeid(Ev).name()).c_str());
162  std::string transitionTag;
163  std::string transitionType;
164 
165  if (typeid(Tag) != typeid(default_transition_name))
166  {
167  transitionTag = demangleSymbol<Tag>();
168  transitionType = getTransitionType<Tag>();
169  ROS_DEBUG_STREAM("TRANSITION TYPE:" << transitionType);
170  }
171  else
172  {
173  transitionTag = "";
174  automaticTransitionTag<Ev>(transitionTag);
175  automaticTransitionType<Ev>(transitionType);
176  }
177 
178  ROS_INFO_STREAM("Transition tag: " << transitionTag);
179 
180  if (!sourceState->stateMachine_->containsState<Dst>())
181  {
182  auto realparentState = sourceState->stateMachine_->getState<typename Dst::TContext>();
183  auto siblingnode = sourceState->stateMachine_->createState<Dst>(realparentState);

185  //auto siblingnode = sourceState->stateMachine_->createState<Dst>(sourceState->parentState_);
186  WalkStatesExecutor<Dst>::walkStates(siblingnode, true);
187  sourceState->declareTransition<Ev>(siblingnode, transitionTag, transitionType, history, transitionTypeInfo);
188  }
189  else

191  //auto realparentState = sourceState->stateMachine_->getState<typename Dst::TContext>();
192  //auto siblingnode = sourceState->stateMachine_->createState<Dst>(realparentState);
193
