149 template <typename T>
150 void Network::tcpReadFromBuffer(std::chrono::microseconds timeout) try {
151  if (!tcp_socket_.poll(timeout.count(), Poco::Net::Socket::SELECT_READ)) {
152  return;
153  }
154 
155  int available_bytes = tcp_socket_.available();
156  if (pending_response_.empty() &&
157  available_bytes >= static_cast<int>(sizeof(typename T::Header))) {
158  typename T::Header header;
159  tcp_socket_.receiveBytes(&header, sizeof(header));
160  if (header.size < sizeof(header)) {
161  throw ProtocolException("libfranka: Incorrect TCP message size.");
162  }
163  pending_response_.resize(header.size);
164  std::memcpy(pending_response_.data(), &header, sizeof(header));
165  pending_response_offset_ = sizeof(header);
166  pending_command_id_ = header.command_id;
167  }
168  if (!pending_response_.empty() && available_bytes > 0) {
169  pending_response_offset_ += tcp_socket_.receiveBytes(
170  &pending_response_[pending_response_offset_],
171  std::min(tcp_socket_.available(),
172  static_cast<int>(pending_response_.size() - pending_response_offset_)));
173  if (pending_response_offset_ == pending_response_.size()) {
174  received_responses_.emplace(pending_command_id_, pending_response_);
175  pending_response_.clear();
176  pending_response_offset_ = 0;
177  pending_command_id_ = 0;
178  }
179  }
180 } catch (const Poco::Exception& e) {
181  using namespace std::string_literals; // NOLINT(google-build-using-namespace)
182  throw NetworkException("libfranka: TCP receive: "s + e.what());
183 }

202 template <typename T>
203 bool Network::tcpReceiveResponse(uint32_t command_id,
204  std::function<void(const typename T::Response&)> handler) {
205  using namespace std::literals::chrono_literals; // NOLINT(google-build-using-namespace)
206  std::unique_lock<std::mutex> lock(tcp_mutex_, std::try_to_lock);
207  if (!lock.owns_lock()) {
208  return false;
209  }
210 
211  tcpReadFromBuffer<T>(0us);
212  decltype(received_responses_)::const_iterator it = received_responses_.find(command_id);
213  if (it != received_responses_.end()) {
214  auto message = reinterpret_cast<const typename T::template Message<typename T::Response>*>(
215  it->second.data());
216  if (message->header.size < sizeof(message)) {
217  throw ProtocolException("libfranka: Incorrect TCP message size.");
218  }
219  handler(message->getInstance());
220  received_responses_.erase(it);
221  return true;
222  }
223  return false;
224 }

227 typename T::Response Network::tcpBlockingReceiveResponse(uint32_t command_id,
228  std::vector<uint8_t>* vl_buffer) {
229  using namespace std::literals::chrono_literals; // NOLINT(google-build-using-namespace)
230  std::unique_lock<std::mutex> lock(tcp_mutex_, std::defer_lock);
231  decltype(received_responses_)::const_iterator it;
232  do {
233  lock.lock();
234  tcpReadFromBuffer<T>(10ms);
235  it = received_responses_.find(command_id);
236  lock.unlock();
237  std::this_thread::yield();
238  } while (it == received_responses_.end());
239 
240  auto message = *reinterpret_cast<const typename T::template Message<typename T::Response>*>(
241  it->second.data());
242  if (message.header.size < sizeof(message)) {
243  throw ProtocolException("libfranka: Incorrect TCP message size.");
244  }
245 
246  if (vl_buffer != nullptr && message.header.size != sizeof(message)) {
247  size_t data_size = message.header.size - sizeof(message);
248  std::vector<uint8_t> data_buffer(data_size);
249  std::memcpy(data_buffer.data(), &it->second[sizeof(message)], data_size);
250  *vl_buffer = data_buffer;
251  }
252 
253  received_responses_.erase(it);
254  return message.getInstance();
255 }
