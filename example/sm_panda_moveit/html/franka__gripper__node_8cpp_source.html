21 void handleErrors(actionlib::SimpleActionServer<T_action>* server,
22  std::function<bool(const T_goal&)> handler,
23  const T_goal& goal) {
24  T_result result;
25  try {
26  result.success = handler(goal);

29  ROS_ERROR_STREAM("" << ex.what());
