108 void CartesianImpedanceExampleController::starting(const ros::Time& /*time*/) {
109  // compute initial velocity with jacobian and set x_attractor and q_d_nullspace
110  // to initial configuration
111  franka::RobotState initial_state = state_handle_->getRobotState();
112  // get jacobian
113  std::array<double, 42> jacobian_array =
114  model_handle_->getZeroJacobian(franka::Frame::kEndEffector);
115  // convert to eigen
116  Eigen::Map<Eigen::Matrix<double, 6, 7>> jacobian(jacobian_array.data());
117  Eigen::Map<Eigen::Matrix<double, 7, 1>> dq_initial(initial_state.dq.data());
118  Eigen::Map<Eigen::Matrix<double, 7, 1>> q_initial(initial_state.q.data());
119  Eigen::Affine3d initial_transform(Eigen::Matrix4d::Map(initial_state.O_T_EE.data()));
120 
121  // set equilibrium point to current state
122  position_d_ = initial_transform.translation();
123  orientation_d_ = Eigen::Quaterniond(initial_transform.linear());
124  position_d_target_ = initial_transform.translation();
125  orientation_d_target_ = Eigen::Quaterniond(initial_transform.linear());
126 
127  // set nullspace equilibrium configuration to initial q
128  q_d_nullspace_ = q_initial;
129 }

131 void CartesianImpedanceExampleController::update(const ros::Time& /*time*/,
132  const ros::Duration& /*period*/) {
133  // get state variables
134  franka::RobotState robot_state = state_handle_->getRobotState();
135  std::array<double, 7> coriolis_array = model_handle_->getCoriolis();
136  std::array<double, 42> jacobian_array =
137  model_handle_->getZeroJacobian(franka::Frame::kEndEffector);
138 
139  // convert to Eigen
140  Eigen::Map<Eigen::Matrix<double, 7, 1>> coriolis(coriolis_array.data());
141  Eigen::Map<Eigen::Matrix<double, 6, 7>> jacobian(jacobian_array.data());
142  Eigen::Map<Eigen::Matrix<double, 7, 1>> q(robot_state.q.data());
143  Eigen::Map<Eigen::Matrix<double, 7, 1>> dq(robot_state.dq.data());
144  Eigen::Map<Eigen::Matrix<double, 7, 1>> tau_J_d( // NOLINT (readability-identifier-naming)
145  robot_state.tau_J_d.data());
146  Eigen::Affine3d transform(Eigen::Matrix4d::Map(robot_state.O_T_EE.data()));
147  Eigen::Vector3d position(transform.translation());
148  Eigen::Quaterniond orientation(transform.linear());
149 
150  // compute error to desired pose
151  // position error
152  Eigen::Matrix<double, 6, 1> error;
153  error.head(3) << position - position_d_;
154 
155  // orientation error
156  if (orientation_d_.coeffs().dot(orientation.coeffs()) < 0.0) {
157  orientation.coeffs() << -orientation.coeffs();
158  }
159  // "difference" quaternion
160  Eigen::Quaterniond error_quaternion(orientation * orientation_d_.inverse());
161  // convert to axis angle
162  Eigen::AngleAxisd error_quaternion_angle_axis(error_quaternion);
163  // compute "orientation error"
164  error.tail(3) << error_quaternion_angle_axis.axis() * error_quaternion_angle_axis.angle();
165 
166  // compute control
167  // allocate variables
168  Eigen::VectorXd tau_task(7), tau_nullspace(7), tau_d(7);
169 
170  // pseudoinverse for nullspace handling
171  // kinematic pseuoinverse
172  Eigen::MatrixXd jacobian_transpose_pinv;
173  pseudoInverse(jacobian.transpose(), jacobian_transpose_pinv);
174 
175  // Cartesian PD control with damping ratio = 1
176  tau_task << jacobian.transpose() *
177  (-cartesian_stiffness_ * error - cartesian_damping_ * (jacobian * dq));
178  // nullspace PD control with damping ratio = 1
179  tau_nullspace << (Eigen::MatrixXd::Identity(7, 7) -
180  jacobian.transpose() * jacobian_transpose_pinv) *
181  (nullspace_stiffness_ * (q_d_nullspace_ - q) -
182  (2.0 * sqrt(nullspace_stiffness_)) * dq);
183  // Desired torque
184  tau_d << tau_task + tau_nullspace + coriolis;
185  // Saturate torque rate to avoid discontinuities
186  tau_d << saturateTorqueRate(tau_d, tau_J_d);
187  for (size_t i = 0; i < 7; ++i) {
188  joint_handles_[i].setCommand(tau_d(i));
189  }
190 
191  // update parameters changed online either through dynamic reconfigure or through the interactive
192  // target by filtering
193  cartesian_stiffness_ =
194  filter_params_ * cartesian_stiffness_target_ + (1.0 - filter_params_) * cartesian_stiffness_;
195  cartesian_damping_ =
196  filter_params_ * cartesian_damping_target_ + (1.0 - filter_params_) * cartesian_damping_;
197  nullspace_stiffness_ =
198  filter_params_ * nullspace_stiffness_target_ + (1.0 - filter_params_) * nullspace_stiffness_;
199  position_d_ = filter_params_ * position_d_target_ + (1.0 - filter_params_) * position_d_;
200  Eigen::AngleAxisd aa_orientation_d(orientation_d_);
201  Eigen::AngleAxisd aa_orientation_d_target(orientation_d_target_);
202  aa_orientation_d.axis() = filter_params_ * aa_orientation_d_target.axis() +
203  (1.0 - filter_params_) * aa_orientation_d.axis();
204  aa_orientation_d.angle() = filter_params_ * aa_orientation_d_target.angle() +
205  (1.0 - filter_params_) * aa_orientation_d.angle();
206  orientation_d_ = Eigen::Quaterniond(aa_orientation_d);
207 }

209 Eigen::Matrix<double, 7, 1> CartesianImpedanceExampleController::saturateTorqueRate(
210  const Eigen::Matrix<double, 7, 1>& tau_d_calculated,
211  const Eigen::Matrix<double, 7, 1>& tau_J_d) { // NOLINT (readability-identifier-naming)
212  Eigen::Matrix<double, 7, 1> tau_d_saturated{};
213  for (size_t i = 0; i < 7; i++) {
214  double difference = tau_d_calculated[i] - tau_J_d[i];
215  tau_d_saturated[i] =
216  tau_J_d[i] + std::max(std::min(difference, delta_tau_max_), -delta_tau_max_);
217  }
218  return tau_d_saturated;
219 }

221 void CartesianImpedanceExampleController::complianceParamCallback(
222  franka_example_controllers::compliance_paramConfig& config,
223  uint32_t /*level*/) {
224  cartesian_stiffness_target_.setIdentity();
225  cartesian_stiffness_target_.topLeftCorner(3, 3)
226  << config.translational_stiffness * Eigen::Matrix3d::Identity();
227  cartesian_stiffness_target_.bottomRightCorner(3, 3)
228  << config.rotational_stiffness * Eigen::Matrix3d::Identity();
229  cartesian_damping_target_.setIdentity();
230  // Damping ratio = 1
231  cartesian_damping_target_.topLeftCorner(3, 3)
232  << 2.0 * sqrt(config.translational_stiffness) * Eigen::Matrix3d::Identity();
233  cartesian_damping_target_.bottomRightCorner(3, 3)
234  << 2.0 * sqrt(config.rotational_stiffness) * Eigen::Matrix3d::Identity();
235  nullspace_stiffness_target_ = config.nullspace_stiffness;
236 }

242  orientation_d_target_.coeffs() << msg->pose.orientation.x, msg->pose.orientation.y,
243  msg->pose.orientation.z, msg->pose.orientation.w;
244  if (last_orientation_d_target.coeffs().dot(orientation_d_target_.coeffs()) < 0.0) {
245  orientation_d_target_.coeffs() << -orientation_d_target_.coeffs();
246  }
