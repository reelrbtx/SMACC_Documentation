108 void OdomTracker::popPath(int items, bool keepPreviousPath)
109 {
110  ROS_INFO("odom_tracker m_mutex acquire");
111  std::lock_guard<std::mutex> lock(m_mutex_);
112 
113  ROS_INFO("POP PATH ENTRY");
114  this->logStateString();
115 
116  if (!keepPreviousPath)
117  {
118  baseTrajectory_.poses.clear();
119  }
120 
121  while (items > 0 && !pathStack_.empty())
122  {
123  auto &stacked = pathStack_.back().poses;
124  baseTrajectory_.poses.insert(baseTrajectory_.poses.begin(), stacked.begin(), stacked.end());
125  pathStack_.pop_back();
126  items--;
127 
128  ROS_INFO("POP PATH Iteration ");
129  this->logStateString();
130  }
131 
132  ROS_INFO("POP PATH EXITING");
133  this->logStateString();
134  ROS_INFO("odom_tracker m_mutex release");
135  this->updateAggregatedStackPath();
136 }

247 bool OdomTracker::updateClearPath(const nav_msgs::Odometry &odom)
248 {
249  // we initially accept any message if the queue is empty
251  geometry_msgs::PoseStamped base_pose;
252 
253  base_pose.pose = odom.pose.pose;
254  base_pose.header = odom.header;
255  baseTrajectory_.header = odom.header;
256 
257  bool acceptBackward = false;
258  bool pullingerror = false;
259  if (baseTrajectory_.poses.empty())
260  {
261  acceptBackward = false;
262  }
263  else
264  {
265  auto &prevPose = baseTrajectory_.poses.back().pose;
266  const geometry_msgs::Point &prevPoint = prevPose.position;
267  double prevAngle = tf::getYaw(prevPose.orientation);
268 
269  auto &currePose = base_pose.pose;
270  const geometry_msgs::Point &currePoint = currePose.position;
271  double currentAngle = tf::getYaw(currePose.orientation);
272 
273  double lastpointdist = p2pDistance(prevPoint, currePoint);
274  double goalAngleOffset = angles::shortest_angular_distance(prevAngle, currentAngle);
275 
276  acceptBackward = !baseTrajectory_.poses.empty() && (lastpointdist > clearPointDistanceThreshold_ || goalAngleOffset > clearAngularDistanceThreshold_);
277 
278  pullingerror = lastpointdist > 2 * clearPointDistanceThreshold_;
279  }
280 
281  //ROS_INFO("Backwards, last distance: %lf < %lf accept: %d", dist, minPointDistanceBackwardThresh_, acceptBackward);
282  if (acceptBackward)
283  {
284  baseTrajectory_.poses.pop_back();
285  }
286  else if (pullingerror)
287  {
288  ROS_INFO_THROTTLE(2, "Incorrect backwards motion.");
289  }
290  else
291  {
293  }
294 
295  return acceptBackward;
296 }
