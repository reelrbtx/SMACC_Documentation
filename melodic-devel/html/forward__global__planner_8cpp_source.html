46 bool ForwardGlobalPlanner::makePlan(const geometry_msgs::PoseStamped &start,
47  const geometry_msgs::PoseStamped &goal, std::vector<geometry_msgs::PoseStamped> &plan)
48 {
49  //ROS_WARN_STREAM("Forward global plan goal: " << goal);
50 
51  //three stages: 1 - heading to goal position, 2 - going forward keep orientation, 3 - heading to goal orientation
52 
53  // 1 - heading to goal position
54  // orientation direction
55 
56  double dx = goal.pose.position.x - start.pose.position.x;
57  double dy = goal.pose.position.y - start.pose.position.y;
58 
59  double lenght = sqrt(dx * dx + dy * dy);
60 
61  geometry_msgs::PoseStamped prevState;
62  if (lenght > skip_straight_motion_distance_)
63  {
64  // skip initial pure spinning and initial straight motion
65  //ROS_INFO("1 - heading to goal position pure spinning");
66  double heading_direction = atan2(dy, dx);
67  prevState = move_base_z_client::makePureSpinningSubPlan(start, heading_direction, plan, puresSpinningRadStep_);
68  //ROS_INFO("2 - going forward keep orientation pure straight");
69  prevState = move_base_z_client::makePureStraightSubPlan(prevState, goal.pose.position, lenght, plan);
70  }
71  else
72  {
73  prevState = start;
74  }
75 
76  //ROS_INFO("3 - heading to goal orientation");
77  double goalOrientation = angles::normalize_angle(tf::getYaw(goal.pose.orientation));
78  move_base_z_client::makePureSpinningSubPlan(prevState, goalOrientation, plan, puresSpinningRadStep_);
79 
80  nav_msgs::Path planMsg;
81  planMsg.poses = plan;
82  planMsg.header.stamp = ros::Time::now();
83  planMsg.header.frame_id = "/odom";
84 
85  // check plan rejection
86  bool acceptedGlobalPlan = true;
87 
88  // static const unsigned char NO_INFORMATION = 255;
89  // static const unsigned char LETHAL_OBSTACLE = 254;
90  // static const unsigned char INSCRIBED_INFLATED_OBSTACLE = 253;
91  // static const unsigned char FREE_SPACE = 0;
92 
93  costmap_2d::Costmap2D *costmap2d = this->costmap_ros_->getCostmap();
94  for (auto &p : plan)
95  {
96  unsigned int mx, my;
97  costmap2d->worldToMap(p.pose.position.x, p.pose.position.y, mx, my);
98  auto cost = costmap2d->getCost(mx, my);
99 
100  if (cost >= costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
101  {
102  acceptedGlobalPlan = false;
103  break;
104  }
105  }
106 
107  if (acceptedGlobalPlan)

109  planPub_.publish(planMsg);
