60 void execute(const sm_dance_bot_2::LEDControlGoalConstPtr& goal) // Note: "Action" is not appended to DoDishes here
61 {
62  ROS_INFO_STREAM("Tool action server request: "<< *goal);
63  cmd = *goal;
64 
65  if(goal->command == sm_dance_bot_2::LEDControlGoal::CMD_ON)
66  {
67  currentState_ = sm_dance_bot_2::LEDControlResult::STATE_RUNNING;
68  }
69  else if (goal->command == sm_dance_bot_2::LEDControlGoal::CMD_OFF)
70  {
71  currentState_ = sm_dance_bot_2::LEDControlResult::STATE_IDLE;
72  }
73 
74  // 10Hz internal loop
75  ros::Rate rate(20);
76 
77  while(ros::ok())
78  {
79  if(as_->isPreemptRequested())
80  {
81  // a new request is being executed, we will stop this one
82  ROS_WARN("LEDActionServer request preempted. Forgetting older request.");
83  as_->setPreempted();
84  return;
85  }
86 
87  publishFeedback();
88  publishStateMarker();
89  rate.sleep();
90  }
91 
92  // never reach succeded because were are interested in keeping the feedback alive
93  as_->setSucceeded();
94 }
