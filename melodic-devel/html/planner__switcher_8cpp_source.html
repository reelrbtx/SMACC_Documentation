72 void PlannerSwitcher::dynreconfCallback(const dynamic_reconfigure::Config::ConstPtr &configuration_update)
73 {

75  [&](const dynamic_reconfigure::StrParameter &p) {
76  return p.name == "base_global_planner" && p.value == desired_global_planner_;
77  });
78 
79  auto lp = std::find_if(configuration_update->strs.begin(), configuration_update->strs.begin(),
80  [&](const dynamic_reconfigure::StrParameter &p) {
81  return p.name == "base_local_planner" && p.value == desired_local_planner_;
82  });
83 
84  if (gp == configuration_update->strs.end() || lp == configuration_update->strs.end())
85  {
86  ROS_INFO("[PlannerSwitcher] After planner update it is noticed an incorrect move_base planner configuration. Resending request.");
