176  virtual double getEffortMax()
177  {
178  /*
179  * gzsdf has a major flaw when using continuous joints. It appears gazebo
180  * cannot handle continuous joints and so it sets the limits to +/-1e16.
181  * This is fine, except it drops the limit effort and limit velocity.
182  * Lack of limit effort causes the robot to implode to the origin if the
183  * controllers are not tuned or experience a disturbance. This little hack
184  * lets us limit the controller effort internally.
185  */
186  if (effort_limit_ < 0.0)
187  return joint_->GetEffortLimit(0);
188  else
189  return effort_limit_;
190  }

208  void update(const ros::Time now, const ros::Duration dt)
209  {
210  actual_velocity_ += 0.1 * (joint_->GetVelocity(0) - actual_velocity_);
211 
212  double effort = 0.0;
213  if (mode_ == MODE_CONTROL_POSITION)
214  {
215  double p_error = angles::shortest_angular_distance(getPosition(), desired_position_);
216  double v = position_pid_.computeCommand(p_error, dt) + desired_velocity_;
217  v = std::min(getVelocityMax(), std::max(-getVelocityMax(), v));
218  double t = velocity_pid_.computeCommand(v - actual_velocity_, dt);
219  effort = t + desired_effort_;
220  }
221  else if (mode_ == MODE_CONTROL_VELOCITY)
222  {
223  double t = velocity_pid_.computeCommand(desired_velocity_ - actual_velocity_, dt);
224  effort = t + desired_effort_;
225  }
226  else if (mode_ == MODE_CONTROL_EFFORT)
227  {
228  effort = desired_effort_;
229  }
230 
231  // Limit effort so robot doesn't implode
232  double lim = getEffortMax();
233  applied_effort_ = std::max(-lim, std::min(effort, lim));
234 
235  // Actually update
236  joint_->SetForce(0, applied_effort_);
237  }
