94 void FetchGazeboPlugin::Init()
95 {
96  // Init time stuff
97  prevUpdateTime = model_->GetWorld()->SimTime();
98  last_publish_ = ros::Time(prevUpdateTime.Double());
99  urdf::Model urdfmodel;
100  if (!urdfmodel.initParam("robot_description"))
101  {
102  ROS_ERROR("Failed to parse URDF");
103  }
104 
105  // Init joint handles
106  gazebo::physics::Joint_V joints = model_->GetJoints();
107  for (physics::Joint_V::iterator it = joints.begin(); it != joints.end(); ++it)
108  {
109  //get effort limit and continuous state from URDF
110  std::shared_ptr<const urdf::Joint> urdf_joint = urdfmodel.getJoint((*it)->GetName());
111 
112  JointHandlePtr handle(new JointHandle(*it,
113  urdf_joint->limits->velocity,
114  urdf_joint->limits->effort,
115  (urdf_joint->type == urdf::Joint::CONTINUOUS)));
116  joints_.push_back(handle);
117  robot_controllers::JointHandlePtr h(handle);
118  controller_manager_.addJointHandle(h);
119  }
120 
121  // Init controllers
122  ros::NodeHandle pnh("~");
123  controller_manager_.init(pnh);
124 
125  // Publish joint states only after controllers are fully ready
126  joint_state_pub_ = nh_.advertise<sensor_msgs::JointState>("joint_states", 10);
127 
128  ROS_INFO("Finished initializing FetchGazeboPlugin");
129 }

131 void FetchGazeboPlugin::OnUpdate(
132  const common::UpdateInfo& info)
133 {
134  if (!ros::ok())
135  return;
136 
137  // Get time and timestep for controllers
138  common::Time currTime = model_->GetWorld()->SimTime();
139  common::Time stepTime = currTime - prevUpdateTime;
140  prevUpdateTime = currTime;
141  double dt = stepTime.Double();
142  ros::Time now = ros::Time(currTime.Double());
143 
144  // Update controllers
145  controller_manager_.update(now, ros::Duration(dt));
146 
147  // Update joints back into Gazebo
148  for (size_t i = 0; i < joints_.size(); ++i)
149  joints_[i]->update(now, ros::Duration(dt));
150 
151  // Limit publish rate
152  if (now - last_publish_ < ros::Duration(0.01))
153  return;
154 
155  // Publish joint_state message
156  sensor_msgs::JointState js;
157  js.header.stamp = ros::Time(currTime.Double());
158  for (size_t i = 0; i < joints_.size(); ++i)
159  {
160  js.name.push_back(joints_[i]->getName());
161  js.position.push_back(joints_[i]->getPosition());
162  js.velocity.push_back(joints_[i]->getVelocity());
163  js.effort.push_back(joints_[i]->getEffort());
164  }
165  joint_state_pub_.publish(js);
166 
167  last_publish_ = now;
168 }
